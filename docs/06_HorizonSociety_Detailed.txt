Critical Implementation Notes
1. Overview & Responsibilities
HorizonSociety implements a fully autonomous simulation of human (and humanoid) civilizations parallel to the player. It replaces static vanilla villages with dynamic, expanding empires that research technologies, wage wars, form alliances, and manage their economies.
Dynamic Civilizations: NPCs group into Civilizations that own multiple Settlements. They advance through HorizonAges independently, potentially outpacing the player.
Government Types: 11 distinct governance models (Tribalism to Corporation) that dictate diplomatic modifiers, military aggression, and internal stability.
Professions & Social Classes: 34 distinct NPC professions organized into a 7-tier social hierarchy. NPCs perform daily schedules and affect settlement resource output.
Diplomacy & Treaties: 15 unique diplomatic actions forming a web of relations between civilizations and the player.
Espionage: Covert operations allowing technology theft, sabotage, and assassination.
AI Director: Evaluates settlement growth, assigns construction goals, coordinates army movement, and dictates strategic responses to player actions.

package com.horizonpack.horizonsociety.core;

import com.horizonpack.horizonsociety.events.SocietyEventHandlers;
import com.horizonpack.horizonsociety.network.SocietyPacketHandler;
import net.neoforged.bus.api.IEventBus;
import net.neoforged.fml.ModContainer;
import net.neoforged.fml.common.Mod;
import net.neoforged.fml.config.ModConfig;
import net.neoforged.fml.event.lifecycle.FMLCommonSetupEvent;
import net.neoforged.neoforge.common.NeoForge;

@Mod(HorizonSociety.MODID)
public class HorizonSociety {
    public static final String MODID = "horizonsociety";
    public static final String VERSION = "1.0.0";

    public HorizonSociety(IEventBus modEventBus, ModContainer modContainer) {
        modContainer.registerConfig(ModConfig.Type.COMMON, SocietyConfig.COMMON_SPEC);
        modContainer.registerConfig(ModConfig.Type.CLIENT, SocietyConfig.CLIENT_SPEC);

        SocietyRegistries.BLOCKS.register(modEventBus);
        SocietyRegistries.ITEMS.register(modEventBus);
        SocietyRegistries.BLOCK_ENTITIES.register(modEventBus);
        SocietyRegistries.ENTITY_TYPES.register(modEventBus);
        SocietyRegistries.MENU_TYPES.register(modEventBus);
        
        SocietyRegistries.GOVERNMENT_TYPES.register(modEventBus);
        SocietyRegistries.PERSONALITY_TYPES.register(modEventBus);
        SocietyRegistries.DIPLOMATIC_ACTIONS.register(modEventBus);
        SocietyRegistries.PROFESSIONS.register(modEventBus);

        modEventBus.addListener(this::commonSetup);
        modEventBus.addListener(SocietyRegistries::registerNewRegistries);
        modEventBus.addListener(SocietyRegistries::registerEntityAttributes);
        modEventBus.addListener(SocietyPacketHandler::register);

        NeoForge.EVENT_BUS.register(SocietyEventHandlers.class);
    }

    private void commonSetup(FMLCommonSetupEvent event) {
        event.enqueueWork(() -> {
            com.horizonpack.horizonsociety.ai.CivilizationDirector.init();
        });
    }
}

package com.horizonpack.horizonsociety.ai;

import com.horizonpack.horizonsociety.state.Civilization;
import com.horizonpack.horizonsociety.state.SpyNetwork;
import net.minecraft.server.level.ServerLevel;

import java.util.Random;
import java.util.UUID;

public class EspionageEngine {
    
    public enum CovertMission { SURVEILLANCE, THEFT, SABOTAGE, ASSASSINATION, PROPAGANDA, CRYPTOGRAPHY }

    public static void executeMission(ServerLevel level, Civilization attacker, Civilization target, CovertMission mission) {
        SpyNetwork net = attacker.getSpyNetwork(target.getId());
        Random rand = level.getRandom();
        
        float successChance = (net.getInfiltrationLevel() * 0.1f) - target.getCounterEspionageLevel();
        successChance = Math.max(0.1f, Math.min(0.9f, successChance));
        
        boolean success = rand.nextFloat() < successChance;
        
        if (success) {
            switch(mission) {
                case SURVEILLANCE -> net.revealIntel(target);
                case THEFT -> target.stealTechnology(attacker);
                case SABOTAGE -> target.destroyRandomInfrastructure(level);
                case ASSASSINATION -> target.killLeader(level);
                case PROPAGANDA -> target.reduceStability(0.1f);
                case CRYPTOGRAPHY -> net.breakCodes();
            }
            net.addInfiltration(1);
        } else {
            // Mission failed, spies caught
            net.reduceInfiltration(5);
            target.decreaseReputation(attacker.getId(), 20); // Diplomatic incident
        }
    }
}

import java.util.UUID;

public class SocietySavedData extends SavedData {
    public static final String DATA_NAME = "horizonsociety_data";

    private final Map<UUID, Civilization> civilizations = new HashMap<>();

    public static SocietySavedData get(ServerLevel level) {
        return level.getDataStorage().computeIfAbsent(
                new SavedData.Factory<>(SocietySavedData::new, SocietySavedData::load),
                DATA_NAME
        );
    }

    public Civilization getCivilization(UUID id) {
        return civilizations.get(id);
    }
    
    public void addCivilization(Civilization civ) {
        civilizations.put(civ.getId(), civ);
        setDirty();
    }
    
    public Map<UUID, Civilization> getAllCivilizations() {
        return civilizations;
    }

    @Override
    public CompoundTag save(CompoundTag tag) {
        ListTag civList = new ListTag();
        for (Civilization civ : civilizations.values()) {
            civList.add(civ.serializeNBT());
        }
        tag.put("Civilizations", civList);
        return tag;
    }

    public static SocietySavedData load(CompoundTag tag) {
        SocietySavedData data = new SocietySavedData();
        ListTag civList = tag.getList("Civilizations", Tag.TAG_COMPOUND);
        for (int i = 0; i < civList.size(); i++) {
            Civilization civ = Civilization.deserializeNBT(civList.getCompound(i));
            data.civilizations.put(civ.getId(), civ);
        }
        return data;
    }
}

package com.horizonpack.horizonsociety.state;

import net.minecraft.nbt.CompoundTag;
import net.minecraft.resources.ResourceLocation;

import java.util.HashMap;

public class Civilization {
    private UUID id;
    private String name;
    private ResourceLocation governmentTypeId;
    private ResourceLocation personalityTypeId;
    private String currentAgeId;
    
    private final Map<UUID, DiplomaticRelation> relations = new HashMap<>();
    private final Map<UUID, SpyNetwork> spyNetworks = new HashMap<>();
    
    private float stability = 1.0f;
    private float counterEspionageLevel = 0.1f;

    public Civilization(UUID id, String name, ResourceLocation govId, ResourceLocation persId, String ageId) {
        this.id = id;
        this.name = name;
        this.governmentTypeId = govId;
        this.personalityTypeId = persId;
        this.currentAgeId = ageId;
    }

    public UUID getId() { return id; }
    public String getName() { return name; }
    public String getCurrentAgeId() { return currentAgeId; }
    public void setCurrentAgeId(String age) { this.currentAgeId = age; }
    
    public DiplomaticRelation getRelation(UUID otherCiv) {
        return relations.computeIfAbsent(otherCiv, k -> new DiplomaticRelation(this.id, k));
    }
    
    public SpyNetwork getSpyNetwork(UUID targetCiv) {
        return spyNetworks.computeIfAbsent(targetCiv, k -> new SpyNetwork(this.id, k));
    }
    
    public void decreaseReputation(UUID otherCiv, int amount) {
        getRelation(otherCiv).adjustReputation(-amount);
    }
    
    public void reduceStability(float amount) {
        this.stability = Math.max(0.0f, this.stability - amount);
    }
    
    public float getCounterEspionageLevel() { return counterEspionageLevel; }

    public void stealTechnology(Civilization from) { /* Integration with HorizonCore */ }
    public void destroyRandomInfrastructure(net.minecraft.server.level.ServerLevel level) { /* Find settlement and break blocks */ }
    public void killLeader(net.minecraft.server.level.ServerLevel level) { this.reduceStability(0.5f); }

    public CompoundTag serializeNBT() {
        CompoundTag tag = new CompoundTag();
        tag.putUUID("Id", id);
        tag.putString("Name", name);
        tag.putString("GovId", governmentTypeId.toString());
        tag.putString("PersId", personalityTypeId.toString());
        tag.putString("AgeId", currentAgeId);
        tag.putFloat("Stability", stability);
        tag.putFloat("CounterEspionage", counterEspionageLevel);

    }

        Civilization civ = new Civilization(
            tag.getUUID("Id"),
            tag.getString("Name"),
            ResourceLocation.parse(tag.getString("GovId")),
            ResourceLocation.parse(tag.getString("PersId")),
            tag.getString("AgeId")
        );
        civ.stability = tag.getFloat("Stability");
        civ.counterEspionageLevel = tag.getFloat("CounterEspionage");
        return civ;
    }

@Mod.EventBusSubscriber(modid = HorizonSociety.MODID)
public class SocietyEventHandlers {

    private static int tickCounter = 0;

    @SubscribeEvent
    public static void onLevelTick(LevelTickEvent.Post event) {
        if (!(event.getLevel() instanceof ServerLevel level)) return;
        
        tickCounter++;
        if (tickCounter >= 1200) { // Check AI Director every 1 minute
            tickCounter = 0;
            CivilizationDirector.tickDirector(level);
        }
    }

    @SubscribeEvent
    public static void onDayAdvance(DayAdvanceEvent event) {
        if (event.level() instanceof ServerLevel level) {
            CivilizationDirector.onDailyTick(level);
        }
    }

    public static final Client CLIENT;

        var commonPair = new ModConfigSpec.Builder().configure(Common::new);
        COMMON = commonPair.getLeft();
        COMMON_SPEC = commonPair.getRight();

26. Critical Implementation Notes
Custom Registry Synchronization: Government types, personality types, professions, and diplomatic actions are implemented using ResourceKey<Registry<T>> initialized in the NewRegistryEvent with .sync(true). This enforces client-server symmetry, guaranteeing that GUI diplomacy screens do not crash due to mismatched ID lookups.
AI Director Thread Offloading: CivilizationDirector.tickDirector() scans massive amounts of block data to update settlement bounding boxes and resource yields. This must be distributed. Accumulate block reads into a queue and process max 50 blocks per tick on the main thread, or dispatch pure calculations to Util.backgroundExecutor().
NBT Depth and Saving: SocietySavedData retains the global map of civilizations. To prevent massive NBT bloat when serializing thousands of NPC relationships, the DiplomaticRelation class serializes exclusively non-neutral relations. If civ A has not met civ B, it does not write to NBT.
Espionage Probability Calculation: The success of an espionage mission inside EspionageEngine dynamically checks target counterEspionageLevel (which scales with Government Type and Police/Guard count). Random checks should use the specific ServerLevel#getRandom() rather than Math.random() to preserve deterministic outcomes for debugging.
NPC Entity Data Synching: The NpcEntity defines its profession via EntityDataAccessor<String>. Due to generic mapping, do not pass bare ResourceLocation directly into the SynchedEntityData; use String and parse it back to a ResourceLocation during getters to avoid classloader serialization issues in 1.21.1.
Diplomacy Packets: RequestDiplomacyPacket is sent from the client's EmbassyScreen. Since players can theoretically spoof packets to force treaties, the handleServer block must retrieve the exact state of SocietySavedData and manually re-evaluate DiplomacyEngine.canAccept(targetCiv, action) before applying the treaty state change.
Social Class Wealth Modifiers: When implementing ProfessionDefinition, the assigned SocialClass carries a wealthMultiplier. HorizonEconomy relies on this multiplier to distribute tax yields when the AI Director calculates daily settlement revenue. Do not nullify this object.
Independent Civ Advancements: Check the civsAdvanceIndependently boolean config parameter during the DayAdvanceEvent inside CivilizationDirector.onDailyTick. If enabled, civs track their own currentAgeId and can unlock new block schemas. If disabled, clamp their currentAgeId to the global HorizonCore WorldAgeData.getGlobalAge().
