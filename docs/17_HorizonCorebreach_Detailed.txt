1. Overview & Responsibilities
HorizonCorebreach represents the grand culmination of the Horizon modpack. After guiding a civilization from the Stone Age into the intergalactic and interdimensional eras, the very fabric of reality begins to unravel due to the extreme harvesting of Ley Lines and dimensional transit.
Reality Corruption: As the timeline progresses into the post-modern age, the world experiences severe localized phenomena, including gravity inversion, time loops, and sky fractures.
The Corebreach: Players must drill into the center of the world (the Core Chamber) using endgame machinery to confront the source of the anomaly.
The Final Encounter: A massive, 7-phase boss fight dynamically altering the environment and challenging every skill, technology, and magic spell the player has acquired.
Multiple Endings: 5 distinct narrative and mechanical resolutions to the game, altering the persistent state of the server permanently based on the player's final choice.

import net.neoforged.fml.ModContainer;
import net.neoforged.fml.common.Mod;
import net.neoforged.fml.config.ModConfig;
import net.neoforged.fml.event.lifecycle.FMLCommonSetupEvent;
import net.neoforged.neoforge.common.NeoForge;

@Mod(HorizonCorebreach.MODID)
public class HorizonCorebreach {
    public static final String MODID = "horizoncorebreach";
    public static final String VERSION = "1.0.0";

    public HorizonCorebreach(IEventBus modEventBus, ModContainer modContainer) {
        modContainer.registerConfig(ModConfig.Type.COMMON, CorebreachConfig.COMMON_SPEC);
        modContainer.registerConfig(ModConfig.Type.CLIENT, CorebreachConfig.CLIENT_SPEC);

        CorebreachRegistries.BLOCKS.register(modEventBus);
        CorebreachRegistries.ITEMS.register(modEventBus);
        CorebreachRegistries.BLOCK_ENTITIES.register(modEventBus);
        CorebreachRegistries.ENTITY_TYPES.register(modEventBus);
        CorebreachRegistries.MENU_TYPES.register(modEventBus);
        CorebreachDataComponents.DATA_COMPONENT_TYPES.register(modEventBus);

        CorebreachRegistries.BOSS_PHASES.register(modEventBus);
        CorebreachRegistries.CORRUPTION_EVENTS.register(modEventBus);
        CorebreachRegistries.ENDGAME_STRUCTURES.register(modEventBus);
        CorebreachRegistries.ENDINGS.register(modEventBus);

        CorebreachAttachments.ATTACHMENT_TYPES.register(modEventBus);

        modEventBus.addListener(this::commonSetup);
        modEventBus.addListener(CorebreachRegistries::registerNewRegistries);
        modEventBus.addListener(CorebreachRegistries::registerEntityAttributes);
        modEventBus.addListener(CorebreachPacketHandler::register);

        NeoForge.EVENT_BUS.register(CorebreachEventHandlers.class);
    }

    private void commonSetup(FMLCommonSetupEvent event) {
        event.enqueueWork(() -> {
            RealityCorruptionEngine.init();
            EndingResolutionEngine.init();
        });
    }
}

package com.horizonpack.horizoncorebreach.data.attachments;

import net.minecraft.nbt.CompoundTag;
import net.neoforged.neoforge.common.util.INBTSerializable;

public class WorldCorruptionData implements INBTSerializable<CompoundTag> {
    private float globalCorruptionLevel = 0.0f;
    private boolean isCoreBreached = false;
    private int activeBossPhase = 0;

    public float getGlobalCorruptionLevel() { return globalCorruptionLevel; }
    public void setGlobalCorruptionLevel(float level) { this.globalCorruptionLevel = Math.max(0, Math.min(100.0f, level)); }
    public void addGlobalCorruptionLevel(float amount) { setGlobalCorruptionLevel(globalCorruptionLevel + amount); }

    public boolean isCoreBreached() { return isCoreBreached; }
    public void setCoreBreached(boolean breached) { this.isCoreBreached = breached; }

    public int getActiveBossPhase() { return activeBossPhase; }
    public void setActiveBossPhase(int phase) { this.activeBossPhase = phase; }

    @Override
    public CompoundTag serializeNBT(net.minecraft.core.HolderLookup.Provider provider) {
        CompoundTag tag = new CompoundTag();
        tag.putFloat("GlobalCorruption", globalCorruptionLevel);
        tag.putBoolean("CoreBreached", isCoreBreached);
        tag.putInt("ActiveBossPhase", activeBossPhase);
        return tag;
    }

    @Override
    public void deserializeNBT(net.minecraft.core.HolderLookup.Provider provider, CompoundTag nbt) {
        this.globalCorruptionLevel = nbt.getFloat("GlobalCorruption");
        this.isCoreBreached = nbt.getBoolean("CoreBreached");
        this.activeBossPhase = nbt.getInt("ActiveBossPhase");
    }
}

import java.util.Random;

public class RealityCorruptionEngine {

    public static void init() {}

    public static void tickCorruption(ServerLevel level) {
        WorldCorruptionData data = level.getData(CorebreachAttachments.WORLD_CORRUPTION);
        
        if (data.getGlobalCorruptionLevel() > 20.0f) {
            Random rand = level.getRandom();
            // 0.1% chance per tick to spawn an event if highly corrupted
            if (rand.nextFloat() < (data.getGlobalCorruptionLevel() / 10000.0f)) {
                triggerRandomEvent(level, data.getGlobalCorruptionLevel());
            }
        }
    }

    private static void triggerRandomEvent(ServerLevel level, float corruptionLevel) {
        Random rand = level.getRandom();
        float roll = rand.nextFloat() * 100.0f;
        
        if (roll < corruptionLevel) {
            // Severity scales with corruption
            if (corruptionLevel > 90.0f && rand.nextBoolean()) {
                applyPhysicsBreakdown(level);
            } else if (corruptionLevel > 70.0f) {
                applyGravityInversion(level);
            } else if (corruptionLevel > 50.0f) {
                applySkyFracture(level);
            } else {
                applyBlockDecay(level);
            }
        }
    }

    private static void applyPhysicsBreakdown(ServerLevel level) {
        // Implementation: Entities fall upwards, blocks drop bedrock instead of items
        for (Player p : level.players()) {
            p.displayClientMessage(net.minecraft.network.chat.Component.translatable("event.corruption.physics_breakdown"), true);
        }
    }

    private static void applyGravityInversion(ServerLevel level) {
        for (Player p : level.players()) {
            // Checked in GravityInversionHelper applied to Entity movement vectors
            GravityInversionHelper.invertPlayer(p, 600); // 30 seconds
            p.displayClientMessage(net.minecraft.network.chat.Component.translatable("event.corruption.gravity_inversion"), true);
        }
    }

    private static void applySkyFracture(ServerLevel level) {
        // Dispatches packet to clients to render the SkyFractureRenderer
        com.horizonpack.horizoncorebreach.network.CorebreachPacketHandler.sendToAll(level.getServer(), new com.horizonpack.horizoncorebreach.network.packets.SyncCorruptionPacket(50.0f));
    }

    private static void applyBlockDecay(ServerLevel level) {
        // Randomly transforms loaded chunk blocks into Void blocks
    }

package com.horizonpack.horizoncorebreach.systems;

import com.horizonpack.horizoncorebreach.data.attachments.CorebreachAttachments;
import com.horizonpack.horizoncorebreach.data.attachments.WorldCorruptionData;
import net.minecraft.server.level.ServerLevel;
import net.minecraft.world.entity.player.Player;

import java.util.Random;

public class GravityInversionHelper {

    // Server-side map of player UUID to remaining inversion ticks
    private static final java.util.Map<java.util.UUID, Integer> invertedPlayers = new java.util.HashMap<>();

    public static void invertPlayer(Player player, int ticks) {
        invertedPlayers.put(player.getUUID(), ticks);
        com.horizonpack.horizoncorebreach.network.CorebreachPacketHandler.sendToPlayer(
            (net.minecraft.server.level.ServerPlayer) player, 
            new com.horizonpack.horizoncorebreach.network.packets.GravityInversionPacket(true)
        );
    }

    public static void tickInversion(ServerLevel level) {
        java.util.Iterator<java.util.Map.Entry<java.util.UUID, Integer>> it = invertedPlayers.entrySet().iterator();
        while (it.hasNext()) {
            java.util.Map.Entry<java.util.UUID, Integer> entry = it.next();
            Player player = level.getPlayerByUUID(entry.getKey());
            
            if (player != null && !player.isSpectator()) {
                int remaining = entry.getValue() - 1;
                
                // Check if player is wearing full Breach Armor (negates inversion)
                boolean isProtected = false; // Simplified check. Full implementation checks all 4 armor slots for BREACH_ARMOR_MATERIAL
                if (player.getInventory().getArmor(0).is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.BREACH_BOOTS.get()) &&
                    player.getInventory().getArmor(1).is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.BREACH_LEGGINGS.get()) &&
                    player.getInventory().getArmor(2).is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.BREACH_CHESTPLATE.get()) &&
                    player.getInventory().getArmor(3).is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.BREACH_HELMET.get())) {
                    isProtected = true;
                }

                if (!isProtected) {
                    // Apply upward force. Vanilla gravity is -0.08 blocks/tick^2. We apply +0.16 to reverse it and fall upwards.
                    player.setDeltaMovement(player.getDeltaMovement().add(0, 0.16, 0));
                    player.hurtMarked = true; // Force client sync of velocity
                }

                if (remaining <= 0) {
                    it.remove();
                    com.horizonpack.horizoncorebreach.network.CorebreachPacketHandler.sendToPlayer(
                        (net.minecraft.server.level.ServerPlayer) player, 
                        new com.horizonpack.horizoncorebreach.network.packets.GravityInversionPacket(false)
                    );
                } else {
                    entry.setValue(remaining);
                }
            } else {
                it.remove(); // Player disconnected or died
            }
        }
    }

public class CorebreachBossEntity extends Monster {

    public static final EntityDataAccessor<Integer> PHASE = SynchedEntityData.defineId(CorebreachBossEntity.class, EntityDataSerializers.INT);

    private final ServerBossEvent bossEvent = (ServerBossEvent) (new ServerBossEvent(this.getDisplayName(), BossEvent.BossBarColor.PURPLE, BossEvent.BossBarOverlay.PROGRESS)).setDarkenScreen(true).setPlayBossMusic(true);
    private int currentPhaseIndex = 0;
    private List<BossPhaseDef> phasesCache;

    public CorebreachBossEntity(EntityType<? extends Monster> type, Level level) {
        super(type, level);
        // Boss is massive and immune to standard knockback/fire
        this.blocksBuilding = true;
        this.xpReward = 50000;
    }

    public static AttributeSupplier.Builder createAttributes() {
        return Monster.createMonsterAttributes()
                .add(Attributes.MAX_HEALTH, 10000.0D) // 10,000 HP
                .add(Attributes.MOVEMENT_SPEED, 0.3D)
                .add(Attributes.ATTACK_DAMAGE, 50.0D)
                .add(Attributes.ARMOR, 20.0D)
                .add(Attributes.ARMOR_TOUGHNESS, 10.0D)
                .add(Attributes.KNOCKBACK_RESISTANCE, 1.0D);
    }

    @Override
    protected void defineSynchedData(SynchedEntityData.Builder builder) {
        super.defineSynchedData(builder);
        builder.define(PHASE, 0);
    }

    @Override
    public void startSeenByPlayer(ServerPlayer player) {
        super.startSeenByPlayer(player);
        this.bossEvent.addPlayer(player);
    }

    @Override
    public void stopSeenByPlayer(ServerPlayer player) {
        super.stopSeenByPlayer(player);
        this.bossEvent.removePlayer(player);
    }

    @Override
    public void tick() {
        super.tick();
        if (!this.level().isClientSide()) {
            this.bossEvent.setProgress(this.getHealth() / this.getMaxHealth());
            checkPhaseTransition();
            executePhaseAttacks();
        }
    }

    private void checkPhaseTransition() {
        if (phasesCache == null) {
            phasesCache = this.level().registryAccess().registryOrThrow(CorebreachRegistries.BOSS_PHASE_KEY).stream().toList();
        }

        if (currentPhaseIndex < phasesCache.size() - 1) {
            BossPhaseDef nextPhase = phasesCache.get(currentPhaseIndex + 1);
            float healthPercent = this.getHealth() / this.getMaxHealth();

            if (healthPercent <= nextPhase.healthThreshold()) {
                BossPhaseDef oldPhase = phasesCache.get(currentPhaseIndex);
                currentPhaseIndex++;
                this.entityData.set(PHASE, currentPhaseIndex);

                // Update Boss Bar color to indicate intensity
                switch(currentPhaseIndex) {
                    case 2: this.bossEvent.setColor(BossEvent.BossBarColor.BLUE); break;
                    case 4: this.bossEvent.setColor(BossEvent.BossBarColor.YELLOW); break;
                    case 6: this.bossEvent.setColor(BossEvent.BossBarColor.RED); break;
                }

                NeoForge.EVENT_BUS.post(new BossPhaseTransitionEvent(this, oldPhase, nextPhase));
            }
        }
    }

    private void executePhaseAttacks() {
        if (phasesCache == null || phasesCache.isEmpty()) return;
        BossPhaseDef currentPhase = phasesCache.get(currentPhaseIndex);
        
        // Attack logic delegates to specialized AI goals or helper methods based on currentPhase.attackPattern()
        // e.g. "summon_minions" spawns RealityFractureElementalEntity instances
        // e.g. "time_dilation" applies intense slowness to players and speeds up the boss
    }

    @Override
    public boolean hurt(DamageSource source, float amount) {
        // Boss is highly resistant to non-endgame weapons
        if (source.getEntity() instanceof net.minecraft.world.entity.player.Player player) {
            net.minecraft.world.item.ItemStack weapon = player.getMainHandItem();
            if (!weapon.is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.REALITY_SWORD.get()) && 
                !weapon.is(com.horizonpack.horizoncorebreach.items.EndgameItemRegistry.VOID_STAFF.get())) {
                amount *= 0.1f; // 90% damage reduction from mundane weapons
            }
        }
        return super.hurt(source, amount);
    }

    @Override
    public void addAdditionalSaveData(CompoundTag tag) {
        super.addAdditionalSaveData(tag);
        tag.putInt("BossPhase", currentPhaseIndex);
    }

    @Override
    public void readAdditionalSaveData(CompoundTag tag) {
        super.readAdditionalSaveData(tag);
        if (tag.contains("BossPhase")) {
            currentPhaseIndex = tag.getInt("BossPhase");
            this.entityData.set(PHASE, currentPhaseIndex);
        }
    }

package com.horizonpack.horizoncorebreach.blockentities;

import com.horizonpack.horizoncorebreach.core.CorebreachRegistries;
import net.minecraft.core.BlockPos;
import net.minecraft.nbt.CompoundTag;
import net.minecraft.world.level.Level;
import net.minecraft.world.level.block.entity.BlockEntity;
import net.minecraft.world.level.block.state.BlockState;

public class RealityAnchorBlockEntity extends BlockEntity {

    private int energyStored = 0;
    private static final int MAX_ENERGY = 1000000;
    private static final int ENERGY_DRAIN_PER_TICK = 500;
    public static final int PROTECTION_RADIUS = 128;

    public RealityAnchorBlockEntity(BlockPos pos, BlockState state) {
        super(CorebreachRegistries.BLOCK_ENTITIES.getEntries().stream()
                .filter(r -> r.getId().getPath().equals("reality_anchor")).findFirst().get().get(), pos, state);
    }

    public static void serverTick(Level level, BlockPos pos, BlockState state, RealityAnchorBlockEntity be) {
        if (be.energyStored >= ENERGY_DRAIN_PER_TICK) {
            be.energyStored -= ENERGY_DRAIN_PER_TICK;
            // Anchor is active. CorruptionEngine checks for nearby active anchors before triggering localized events.
            be.setChanged();
        }
    }

    public int getEnergyStored() { return energyStored; }
    public void receiveEnergy(int amount) {
        this.energyStored = Math.min(MAX_ENERGY, this.energyStored + amount);
        setChanged();
    }

    @Override
    protected void saveAdditional(CompoundTag tag, net.minecraft.core.HolderLookup.Provider registries) {
        super.saveAdditional(tag, registries);
        tag.putInt("Energy", energyStored);
    }

    @Override
    protected void loadAdditional(CompoundTag tag, net.minecraft.core.HolderLookup.Provider registries) {
        super.loadAdditional(tag, registries);
        this.energyStored = tag.getInt("Energy");
    }
}

import net.minecraft.server.level.ServerLevel;
import net.neoforged.bus.api.SubscribeEvent;
import net.neoforged.fml.common.Mod;
import net.neoforged.neoforge.event.tick.LevelTickEvent;

@Mod.EventBusSubscriber(modid = HorizonCorebreach.MODID)
public class CorebreachEventHandlers {

    private static int tickCounter = 0;

    @SubscribeEvent
    public static void onLevelTick(LevelTickEvent.Post event) {
        if (!(event.getLevel() instanceof ServerLevel level)) return;

        tickCounter++;
        
        // Run Gravity Inversion every tick for smooth movement
        GravityInversionHelper.tickInversion(level);

        // Run heavy corruption checks every 20 ticks (1 second)
        if (tickCounter >= 20) {
            tickCounter = 0;
            RealityCorruptionEngine.tickCorruption(level);
            
            // Slowly increase global corruption if the core is breached and boss isn't dead
            WorldCorruptionData data = level.getData(CorebreachAttachments.WORLD_CORRUPTION);
            if (data.isCoreBreached()) {
                data.addGlobalCorruptionLevel(0.001f);
            }
        }
    }
}

package com.horizonpack.horizoncorebreach.client.render;

import com.mojang.blaze3d.systems.RenderSystem;
import com.mojang.blaze3d.vertex.PoseStack;
import net.minecraft.client.Minecraft;
import net.minecraft.client.renderer.GameRenderer;
import net.minecraft.resources.ResourceLocation;
import net.neoforged.neoforge.client.event.RenderLevelStageEvent;
import net.neoforged.bus.api.SubscribeEvent;

public class SkyFractureRenderer {

    private static final ResourceLocation FRACTURE_TEXTURE = ResourceLocation.fromNamespaceAndPath("horizoncorebreach", "textures/environment/sky_fracture.png");
    public static boolean isFractured = false;

    @SubscribeEvent
    public static void onRenderSky(RenderLevelStageEvent event) {
        if (event.getStage() != RenderLevelStageEvent.Stage.AFTER_SKY) return;
        if (!isFractured) return;

        Minecraft mc = Minecraft.getInstance();
        PoseStack poseStack = event.getPoseStack();

        RenderSystem.setShader(GameRenderer::getPositionTexShader);
        RenderSystem.setShaderTexture(0, FRACTURE_TEXTURE);
        RenderSystem.enableBlend();
        RenderSystem.defaultBlendFunc();
        RenderSystem.depthMask(false);

        poseStack.pushPose();
        // Render a massive sphere/dome around the player with the fracture texture, slowly rotating
        // (Vertex buffer generation omitted for brevity)
        poseStack.popPose();

        RenderSystem.depthMask(true);
        RenderSystem.disableBlend();
    }
}

27. Critical Implementation Notes
Gravity Inversion Execution: Vanilla Minecraft does not natively support inverted gravity. The GravityInversionHelper manages this by adding an upward vector (0.16) to the player's deltaMovement every tick, counteracting the hardcoded -0.08 gravity drop. Players must be flagged with hurtMarked = true to force the server to accept the new Y velocity and prevent client-side anti-cheat rubberbanding.
Boss Phase Transitions: The CorebreachBossEntity caches the list of BossPhaseDef objects from the registry upon spawning. Checking thresholds dynamically (this.getHealth() / this.getMaxHealth() <= nextPhase.healthThreshold()) ensures that massive bursts of damage correctly skip intermediate phases if necessary, though the AI logic must handle firing queued environment effects.
Global Corruption Sync: The WorldCorruptionData attachment is tied to the ServerLevel. Because it dictates the probability of severe events, it must be synced to clients via SyncCorruptionPacket whenever it increments by a whole number, allowing the CorruptionOverlay to render screen-edge vignettes or static artifacts smoothly.
Reality Anchor Network: The RealityAnchorBlockEntity requires power from HorizonPower. To efficiently protect areas without scanning all chunks for anchors during a corruption event, anchors register their BlockPos into a static spatial hashmap in RealityCorruptionEngine upon placement and deregister upon destruction or power failure.
Armor Set Bonus: The BREACH_ARMOR_MATERIAL provides immunity to specific corruption events (like core_pulse and gravity_inversion). This requires checking all four armor slots during the event execution loops. Use player.getInventory().getArmor(index) mapped to the specific DeferredItem<ArmorItem> constants.
Boss Event Bar: CorebreachBossEntity utilizes ServerBossEvent to display its health globally. When the phase index changes, the bar's color is explicitly mutated (e.g., Purple -> Blue -> Yellow -> Red) to visually communicate escalating threat to players without requiring them to read chat messages.
Time Loop Event: The time_loop corruption event records the player's exact XYZ coordinates, inventory state, and health when it triggers. After the 1200 tick duration expires, TimeLoopHandler violently resets the player to that exact cached state, simulating a chronological stutter.
Ending Resolution Persistence: When an ending is triggered (e.g., destroy_the_core), the EndingResolutionEngine writes a permanent flag to the WorldCorruptionData NBT. Endings flagged as isGameEnding = true trigger MinecraftServer#halt(false) after playing a cutscene or displaying a final message, effectively concluding that world's narrative loop.
