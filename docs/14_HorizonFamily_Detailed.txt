1. Overview & Responsibilities
HorizonFamily introduces a deep, generational simulation to Minecraft, governing how entities (both players and NPCs) interact, form bonds, age, and pass on their legacy.
Generational Aging: Entities progress through defined life stages, gaining stat modifications and eventual mortality.
Dynasties & Lineage: Complete tracking of family trees, allowing players to view their ancestors and descendants via the Genealogy Table.
Inheritance System: Mechanisms for passing items, wealth, and titles to designated heirs upon death.
Social Standing & Reputation: Players and NPCs possess dynamic reputation scores that unlock hierarchical Social Titles, affecting diplomacy and trade.
Interpersonal Relationships: A complex web of 21 distinct relationship types tracking the exact socio-political bonds between any two entities.
Family Events: Dynamic life milestones (births, marriages, feuds) that permanently alter reputation and relationship standings.

import net.neoforged.fml.config.ModConfig;
import net.neoforged.fml.event.lifecycle.FMLCommonSetupEvent;
import net.neoforged.neoforge.common.NeoForge;

@Mod(HorizonFamily.MODID)
public class HorizonFamily {
    public static final String MODID = "horizonfamily";
    public static final String VERSION = "1.0.0";

    public HorizonFamily(IEventBus modEventBus, ModContainer modContainer) {
        modContainer.registerConfig(ModConfig.Type.COMMON, FamilyConfig.COMMON_SPEC);
        modContainer.registerConfig(ModConfig.Type.CLIENT, FamilyConfig.CLIENT_SPEC);

        FamilyRegistries.BLOCKS.register(modEventBus);
        FamilyRegistries.ITEMS.register(modEventBus);
        FamilyRegistries.BLOCK_ENTITIES.register(modEventBus);
        FamilyRegistries.MENU_TYPES.register(modEventBus);

        FamilyRegistries.RELATIONSHIP_TYPES.register(modEventBus);
        FamilyRegistries.LIFE_STAGES.register(modEventBus);
        FamilyRegistries.SOCIAL_TITLES.register(modEventBus);
        FamilyRegistries.FAMILY_EVENTS.register(modEventBus);

        FamilyAttachments.ATTACHMENT_TYPES.register(modEventBus);

        modEventBus.addListener(this::commonSetup);
        modEventBus.addListener(FamilyRegistries::registerNewRegistries);
        modEventBus.addListener(FamilyPacketHandler::register);

        NeoForge.EVENT_BUS.register(FamilyEventHandlers.class);
    }

    private void commonSetup(FMLCommonSetupEvent event) {
        event.enqueueWork(() -> {
            AgingEngine.init();
            SocialEngine.init();
            InheritanceEngine.init();
        });
    }
}

package com.horizonpack.horizonfamily.data.attachments;

import net.minecraft.nbt.CompoundTag;
import net.minecraft.nbt.ListTag;
import net.minecraft.nbt.Tag;
import net.minecraft.resources.ResourceLocation;
import net.neoforged.neoforge.common.util.INBTSerializable;

import java.util.HashMap;
import java.util.Map;
import java.util.UUID;

public class SocialData implements INBTSerializable<CompoundTag> {
    private int reputation = 0;
    private int wealth = 0; // Cached from HorizonEconomy
    private ResourceLocation currentTitle = ResourceLocation.fromNamespaceAndPath("horizonfamily", "peasant");
    private final Map<UUID, ResourceLocation> relationships = new HashMap<>();

    public int getReputation() { return reputation; }
    public void setReputation(int rep) { this.reputation = rep; }
    public void addReputation(int amount) { this.reputation += amount; }

    public int getWealth() { return wealth; }
    public void setWealth(int wealth) { this.wealth = wealth; }

    public ResourceLocation getCurrentTitle() { return currentTitle; }
    public void setCurrentTitle(ResourceLocation title) { this.currentTitle = title; }

    public Map<UUID, ResourceLocation> getRelationships() { return relationships; }
    public void setRelationship(UUID target, ResourceLocation relId) { this.relationships.put(target, relId); }
    public ResourceLocation getRelationship(UUID target) { return this.relationships.getOrDefault(target, ResourceLocation.fromNamespaceAndPath("horizonfamily", "stranger")); }

    @Override
    public CompoundTag serializeNBT(net.minecraft.core.HolderLookup.Provider provider) {
        CompoundTag tag = new CompoundTag();
        tag.putInt("Reputation", reputation);
        tag.putInt("Wealth", wealth);
        tag.putString("Title", currentTitle.toString());

        ListTag list = new ListTag();
        for (Map.Entry<UUID, ResourceLocation> entry : relationships.entrySet()) {
            CompoundTag rel = new CompoundTag();
            rel.putUUID("Target", entry.getKey());
            rel.putString("Type", entry.getValue().toString());
            list.add(rel);
        }
        tag.put("Relationships", list);

        return tag;
    }

    @Override
    public void deserializeNBT(net.minecraft.core.HolderLookup.Provider provider, CompoundTag nbt) {
        this.reputation = nbt.getInt("Reputation");
        this.wealth = nbt.getInt("Wealth");
        this.currentTitle = ResourceLocation.parse(nbt.getString("Title"));

        this.relationships.clear();
        ListTag list = nbt.getList("Relationships", Tag.TAG_COMPOUND);
        for (int i = 0; i < list.size(); i++) {
            CompoundTag rel = list.getCompound(i);
            this.relationships.put(rel.getUUID("Target"), ResourceLocation.parse(rel.getString("Type")));
        }
    }
}

package com.horizonpack.horizonfamily.data.attachments;

import net.minecraft.nbt.CompoundTag;
import net.minecraft.nbt.ListTag;
import net.minecraft.nbt.Tag;
import net.neoforged.neoforge.common.util.INBTSerializable;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

public class LineageData implements INBTSerializable<CompoundTag> {
    private UUID dynastyId = UUID.randomUUID();
    private String familyName = "Unknown";
    private int ageInDays = 0;
    private ResourceLocation currentLifeStage = ResourceLocation.fromNamespaceAndPath("horizonfamily", "young_adult");

    private UUID spouse = null;
    private UUID mother = null;
    private UUID father = null;
    private final List<UUID> children = new ArrayList<>();
    private UUID designatedHeir = null;

    public UUID getDynastyId() { return dynastyId; }
    public void setDynastyId(UUID id) { this.dynastyId = id; }

    public String getFamilyName() { return familyName; }
    public void setFamilyName(String name) { this.familyName = name; }

    public int getAgeInDays() { return ageInDays; }
    public void setAgeInDays(int days) { this.ageInDays = days; }

    public ResourceLocation getCurrentLifeStage() { return currentLifeStage; }
    public void setCurrentLifeStage(ResourceLocation stage) { this.currentLifeStage = stage; }

    public UUID getSpouse() { return spouse; }
    public void setSpouse(UUID spouse) { this.spouse = spouse; }

    public UUID getMother() { return mother; }
    public void setMother(UUID mother) { this.mother = mother; }

    public UUID getFather() { return father; }
    public void setFather(UUID father) { this.father = father; }

    public List<UUID> getChildren() { return children; }
    public void addChild(UUID child) { if (!this.children.contains(child)) this.children.add(child); }

    public UUID getDesignatedHeir() { return designatedHeir; }
    public void setDesignatedHeir(UUID heir) { this.designatedHeir = heir; }

    @Override
    public CompoundTag serializeNBT(net.minecraft.core.HolderLookup.Provider provider) {
        CompoundTag tag = new CompoundTag();
        tag.putUUID("Dynasty", dynastyId);
        tag.putString("FamilyName", familyName);
        tag.putInt("AgeDays", ageInDays);
        tag.putString("LifeStage", currentLifeStage.toString());

        if (spouse != null) tag.putUUID("Spouse", spouse);
        if (mother != null) tag.putUUID("Mother", mother);
        if (father != null) tag.putUUID("Father", father);
        if (designatedHeir != null) tag.putUUID("Heir", designatedHeir);

        ListTag kids = new ListTag();
        for (UUID child : children) {
            CompoundTag cTag = new CompoundTag();
            cTag.putUUID("ID", child);
            kids.add(cTag);
        }
        tag.put("Children", kids);

        return tag;
    }

    @Override
    public void deserializeNBT(net.minecraft.core.HolderLookup.Provider provider, CompoundTag nbt) {
        this.dynastyId = nbt.getUUID("Dynasty");
        this.familyName = nbt.getString("FamilyName");
        this.ageInDays = nbt.getInt("AgeDays");
        this.currentLifeStage = ResourceLocation.parse(nbt.getString("LifeStage"));

        this.spouse = nbt.contains("Spouse") ? nbt.getUUID("Spouse") : null;
        this.mother = nbt.contains("Mother") ? nbt.getUUID("Mother") : null;
        this.father = nbt.contains("Father") ? nbt.getUUID("Father") : null;
        this.designatedHeir = nbt.contains("Heir") ? nbt.getUUID("Heir") : null;

        this.children.clear();
        ListTag kids = nbt.getList("Children", Tag.TAG_COMPOUND);
        for (int i = 0; i < kids.size(); i++) {
            this.children.add(kids.getCompound(i).getUUID("ID"));
        }
    }

import com.horizonpack.horizoncore.events.custom.DayAdvanceEvent;

import net.neoforged.bus.api.SubscribeEvent;
import net.neoforged.fml.common.Mod;

import java.util.Optional;

@Mod.EventBusSubscriber(modid = "horizonfamily")
public class AgingEngine {

    private static final ResourceLocation HEALTH_MOD_ID = ResourceLocation.fromNamespaceAndPath("horizonfamily", "aging_health");
    private static final ResourceLocation SPEED_MOD_ID = ResourceLocation.fromNamespaceAndPath("horizonfamily", "aging_speed");

    public static void init() {}

    @SubscribeEvent
    public static void onDayAdvance(DayAdvanceEvent event) {
        ServerLevel level = event.level();
        for (Player player : level.players()) {
            tickAging(player);
        }
        // NPCs are handled by HorizonSociety which iterates them directly and calls tickAging
    }

    public static void tickAging(LivingEntity entity) {
        if (!entity.hasData(FamilyAttachments.LINEAGE_DATA)) return;

        LineageData lineage = entity.getData(FamilyAttachments.LINEAGE_DATA);
        lineage.setAgeInDays(lineage.getAgeInDays() + 1);

        evaluateLifeStage(entity, lineage);
    }

    private static void evaluateLifeStage(LivingEntity entity, LineageData lineage) {
        int age = lineage.getAgeInDays();
        ResourceLocation currentStageId = lineage.getCurrentLifeStage();

        for (var entry : FamilyRegistries.LIFE_STAGES.getEntries()) {
            LifeStage stage = entry.getValue();
            if (age >= stage.minAgeDays() && age <= stage.maxAgeDays()) {
                if (!currentStageId.equals(entry.getKey().location())) {
                    lineage.setCurrentLifeStage(entry.getKey().location());
                    applyLifeStageModifiers(entity, stage);
                    
                    if (entity instanceof Player player) {
                        player.displayClientMessage(net.minecraft.network.chat.Component.translatable("message.horizonfamily.stage_advance", 
                                net.minecraft.network.chat.Component.translatable(stage.nameKey())), true);
                    }
                }
                break;
            }
        }
    }

    private static void applyLifeStageModifiers(LivingEntity entity, LifeStage stage) {
        var healthInstance = entity.getAttribute(Attributes.MAX_HEALTH);
        if (healthInstance != null) {
            healthInstance.removeModifier(HEALTH_MOD_ID);
            if (stage.healthModifier() != 0.0) {
                healthInstance.addPermanentModifier(new AttributeModifier(HEALTH_MOD_ID, stage.healthModifier(), AttributeModifier.Operation.ADD_MULTIPLIED_TOTAL));
            }
        }

        var speedInstance = entity.getAttribute(Attributes.MOVEMENT_SPEED);
        if (speedInstance != null) {
            speedInstance.removeModifier(SPEED_MOD_ID);
            if (stage.speedModifier() != 0.0) {
                speedInstance.addPermanentModifier(new AttributeModifier(SPEED_MOD_ID, stage.speedModifier(), AttributeModifier.Operation.ADD_MULTIPLIED_TOTAL));
            }
        }
    }
}

import net.minecraft.resources.ResourceLocation;
import net.minecraft.server.level.ServerPlayer;

public class SocialEngine {

    public static void init() {}

    public static void addReputation(ServerPlayer player, int amount) {
        SocialData social = player.getData(FamilyAttachments.SOCIAL_DATA);
        social.addReputation(amount);
        evaluateTitles(player, social);
    }

    private static void evaluateTitles(ServerPlayer player, SocialData social) {
        int currentRep = social.getReputation();
        int currentWealth = social.getWealth();

        ResourceLocation bestTitle = ResourceLocation.fromNamespaceAndPath("horizonfamily", "peasant");
        int highestRepReq = -999;

        for (var entry : FamilyRegistries.SOCIAL_TITLES.getEntries()) {
            SocialTitle title = entry.getValue();
            if (currentRep >= title.requiredReputation() && currentWealth >= title.requiredWealth()) {
                if (title.requiredReputation() > highestRepReq) {
                    highestRepReq = title.requiredReputation();
                    bestTitle = entry.getKey().location();
                }
            }
        }

        if (!social.getCurrentTitle().equals(bestTitle)) {
            social.setCurrentTitle(bestTitle);
            player.displayClientMessage(net.minecraft.network.chat.Component.translatable("message.horizonfamily.title_granted", 
                    net.minecraft.network.chat.Component.translatable(FamilyRegistries.SOCIAL_TITLES.get(bestTitle).nameKey())), false);
        }
    }
}

25. Critical Implementation Notes
NeoForge Attachment Integration: Unlike the old Forge Capabilities system utilized by horizoncore previously, the social and lineage data here leverages the new NeoForge ATTACHMENT_TYPES registry. SocialData and LineageData are attached directly to the Entity object. The builder utilizes .copyOnDeath() so that the player's dynastic lineage and social reputation are not inexplicably wiped during standard non-hardcore deaths.
Aging Sync with World Ticks: The AgingEngine hooks into horizoncore's DayAdvanceEvent rather than running a manual tick counter. This ensures that the entity's physical age perfectly aligns with the passing of civilization eras. If a server administrator skips days via commands, entities will correctly age the exact number of skipped days.
Cross-Mod Inheritance Execution: The InheritanceEngine.triggerInheritance() method runs entirely on the server. Because transferring wealth relies on HorizonEconomy and land/chest rights rely on HorizonSociety, this method must dispatch custom capability checks to those mods safely without crashing if the designated heir is offline or unloaded. Offline heirs receive their inheritance buffered into NBT upon their next login event.
Attribute Modifiers & Lifecycle Safety: The attribute modifiers attached by LifeStage advancement (e.g., -0.6 health modifier for ancient) use specific ResourceLocation UUID equivalents. We must explicitly removeModifier(HEALTH_MOD_ID) before adding the new one in AgingEngine#applyLifeStageModifiers() so they do not infinitely stack when recalculating life stages.
UUID Reference Validations: LineageData stores family members strictly as UUIDs. When attempting to fetch the name of a spouse or parent for the GUI, the client must request SyncLineagePacket payload data because the referenced entity may not be cached locally or may be permanently dead/despawned. The server provides cached String representations of dead ancestors via a secondary lookup map.
Reputation Recalculation Thresholds: Checking social title promotion across 17 distinct titles on every reputation change is computationally cheap but logically complex because players can drop below requirements due to a FamilyEvent like a divorce or feud. SocialEngine evaluates from highest tier down and explicitly checks both reputation AND wealth (pulled dynamically from HorizonEconomy).
Wedding Ring Interaction Logic: When proposing via WeddingRingItem, the item checks the interaction target's SocialData. If the target's relationship tier is not romantic_interest or partner, the proposal defaults to immediate failure (unless bypassed by extreme wealth/social titles like king). If successful, the ring item is consumed and both parties' LineageData spouse UUIDs are mutually populated.
Scalable Data Packs for Relationships: By using custom RegistryKey elements with .sync(true), data pack creators can inject entirely new relationship types (e.g., "Blood Brother" or "Business Partner") without modifying the Java code. Sub-mods can fetch these dynamic relationships using standard BuiltInRegistries.get() patterns. Ensure that any custom relation properly parses the isFamilial boolean to avoid breaking incest-prevention checks in the MarriageEngine.
